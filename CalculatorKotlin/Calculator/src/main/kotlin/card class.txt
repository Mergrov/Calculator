    private data class Card(val rank: Rank, val suit: Suit) {
        override fun toString() = "${rank.symbol}${suit.symbol}"
    }

    companion object {
        private val allCards = buildList {
            Suit.values().forEach { suit ->
                Rank.values().forEach { rank ->
                    add(Card(rank, suit))
                }
            }
        }
    }
